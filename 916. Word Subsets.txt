from collections import Counter
class Solution(object):
    def wordSubsets(self, A, B):
        ans =[]
        d = {}
        res  = True
        for char in B:
            b = Counter(char)
            for key,val in b.items():
                if key not in d:
                    d[key] = val
                else:
                    d[key]   = max(val,d[key])
                    
        for word in A:
            a  = Counter(word)
            for key,val in d.items():
                if key not in a:
                    res = False
                    break
                elif a[key]<val:
                    res = False
                    break
            #print (res)
            if res:
                ans.append(word)
            res= True
        return ans
